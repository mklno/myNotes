5. Constructor functions and the new operator

  • A new empty object is created
  • The function is called and the this keyword is updated to the empty object
  • The newly created object is linked to the prototype
  • The function automatically returns the object which was created in the beginning
  • To check the instance, use instanceOf jonas instanceOf Person returns a boolean
  • To declare methods don't create inside the constructor function, since it will create a copy of the function for all the objects which affects the performance
const Person = function (firstName, birthYear) {
  this.firstName = firstName;
  this.birthYear = birthYear;
  // avoid creating a method like this  
  this.calcAge = function () {
    this.age = 2024 - this.birthYear;
  };
};

8. Prototyplal Inheritance on build in objects

  • Prototype chain in objects and class
console.log(mano.__proto__); // prototype of the class
console.log(mano.__proto__.__proto__); // prototype of the object on the top level
console.log(mano.__proto__.__proto__.__proto__); // returns null
console.log(Person.prototype); // prototype of the class
console.log(mano.hasOwnProperty('firstName')); // hasOwnProperty is from the prototype of the top level object due to prototype chain
  • Prototype chain in arrays
const arr = [2,3,4,5,6,7];
console.log(arr.__proto__);
console.log(Array.prototype);
console.group(arr.__proto__ === Array.prototype);
  • Prototype chain in an DOM element
const h1 = document.querySelector('h1');
console.log(h1.__proto__); // HTML Heading Element
console.log(h1.__proto__.__proto__); // HTML Element
console.log(h1.__proto__.__proto__.__proto__); // Element
console.log(h1.__proto__.__proto__.__proto__.__proto__); // Node
console.log(h1.__proto__.__proto__.__proto__.__proto__.__proto__); // Event Target
console.log(h1.__proto__.__proto__.__proto__.__proto__.__proto__.__proto__); // Object at the top
console.log(h1.__proto__.__proto__.__proto__.__proto__.__proto__.__proto__.__proto__); // null

  • Classes are not hoised and hence they should be defined before creating an instance.
  • Classes are also treated as first class functions and hence they can be passed to other functions and returned by other functions. That's possible because classes are just a different type of functions behind the scenes.
  • Classes are by default executed in strict mode.

11. Static Methods

  • The methods are implemented on the object's constructor rather than instance of the object.
class Person {
  constructor(firstName, birthYear) {
    this.firstName = firstName;
    this.birthYear = birthYear;
  }
  calcAge() {
    return 2024 - this.birthYear;
  }
  set firstName(name){
    this._firstName = name;
  }
  get firstName(){
    return this._firstName;
  }
}

Person.hey = function(){
  console.log("Hello There");
  console.log(this); // this keyword will be point out to the object calling the method as always
}

Person.hey(); // it works since its defined on the constructor
const mano = new Person('Mano', 2023);
mano.hey(); // function is not defined in the prototype of the Person class and hence it cannot inherit it

12. Object.create

  • It works in a pretty different way than constructor functions and classes.
  • We do have the concept of prototypal inheritance here as well however there are no prototype properties involved, also no constructor functions and no new operator.
  • Object.create can be used to manually set the prototype of an object to any other object we want.
const PersonProto = {
  calcAge() {
    console.log(2024 - this.birthYear);
  },
};

const mano = Object.create(PersonProto);
mano.birthYear = 2000;
mano.calcAge();


const PersonProto = {
  calcAge() {
    console.log(2024 - this.birthYear);
  },
  // the proper way of defining a variable inside the prototype
  init(firstName, birthYear){
    this.firstName = firstName;
    this.birthYear = birthYear;
  }
};

const mano = Object.create(PersonProto);
mano.init('Mano', 2000);
mano.calcAge();

13. Inheritance between Classes: Constructor's functions

// Person class
const Person = function (firstName, birthYear) {
  this.firstName = firstName;
  this.birthYear = birthYear;
};

Person.prototype.calcAge = function () {
  return 2000 - this.birthYear;
};

// Student class
const Student = function (firstName, birthYear, course) {
  Person.call(this, firstName, birthYear);
  this.course = course;
};

Student.prototype.intro = function () {
  console.log(`hello I am ${this.firstName}`);
};

const mano = new Student('mano', 2024, 'cs');
console.log(mano);
  • We can create inheritance using the object.create property
Student.prototype = Object.create(Person.prototype);


const Car = function (make, speed) {
  this.make = make;
  this.speed = speed;
};
Car.prototype.accelerate = function () {
  return (this.speed += 10);
};
Car.prototype.brake = function () {
  return (this.speed -= 5);
};
const EV = function (make, speed, charge) {
  Car.call(this, make, speed);
  this.charge = charge;
};

EV.prototype = Object.create(Car.prototype);
EV.prototype.chargeBattery = function (chargeTo) {
  return (this.charge = chargeTo);
};
EV.prototype.accelerate = function () {
  this.speed += 20;
  this.charge -= 1;
};
const tesla = new EV('Tesla', 120, 100);
tesla.accelerate();
tesla.accelerate();
tesla.brake();
console.log(tesla); // { "make": "Tesla", "speed": 155, "charge": 98 }

20. Sample class

class CarCl {
  constructor(make, speed) {
    this.make = make;
    this.speed = speed;
  }

  accelerate() {
    this.speed += 10;
  }

  brake() {
    this.speed -= 5;
  }
}

class EVCl extends CarCl {
  #charge;
  constructor(make, speed, charge) {
    super(make, speed);
    this.#charge = charge;
  }

  accelerate() {
    this.speed += 20;
    this.#charge -= 1;
    return this;
  }

  brake() {
    this.speed -= 10;
    return this;
  }
}

const tesla = new EVCl('Tesla', 120, 23);
// tesla.accelerate();
// console.log(tesla);
// tesla.accelerate();
// tesla.brake();
// console.log(tesla);
tesla.accelerate().accelerate().brake();
console.log(tesla);
