2.1. Rendering the Root Component and Strict Mode

import React from "react";
import ReactDOM from "react-dom/client";

function App() {
  return <div>Hello React</div>;
}

const root = ReactDOM.createRoot(document.getElementById("root"));
root.render(<App />);
  • Using strict mode just renders the components twice.
import React from "react";
import ReactDOM from "react-dom/client";

function App() {
  return <div>Hello React</div>;
}

const root = ReactDOM.createRoot(document.getElementById("root"));
root.render(
  <React.StrictMode>
    <App />
  </React.StrictMode>
);

2.3. Creating and Reusing Components

  • In React, we can create components as a function and then reuse it as many time as we want.
function App() {
  return (
    <div>
      <h1>Hello React</h1>
      <Pizza />
      <Pizza />
    </div>
  );
}

function Pizza() {
  return <h2>hello</h2>;
}

2.7. Passing and Receiving Props

  • Props are being passed
<Pizza
        name="Spinach Pizza"
        ingredients="Tomato, mozarella, ham, aragula, and burrata cheese"
        photoName="pizzas/prosciutto.jpg"
        price="10"
      />
  • Props are being received
  • The props are destructured in the function arguments
function Pizza({ name, ingredients, photoName, price }) {
  return (
    <div>
      <img src={photoName} alt={name}></img>
      <h3>{name}</h3>
      <p>{ingredients}</p>
      <p>{price}</p>
    </div>
  );
}
