1. What is a Doubly Linked List

  • A Doubly Linked List is a data structure similar to Singly Linked List which offers traversal in both the directions.

2. Represenation of a Doubly Linked List

class Node{
    public:
    int data;
    Node* next;
    Node* prev;

    public:
    Node(int data_, Node* next_, Node* prev_){
        data = data_;
        next = next_;
        prev = prev_;
    }

    public:
    Node(int data_){
        data = data_;
        next = NULL;
        prev = NULL;
    }
}

3. Doubly Linked List to an Array

class Solution {
public:
    ListNode* arrayToLinkedList(vector<int> &nums) {
        if (nums.empty()) return nullptr;
        ListNode* head = new ListNode(nums[0]);
        ListNode* prev = head;
        for (int i = 1; i < nums.size(); ++i) {
            ListNode* el = new ListNode(nums[i],NULL, prev);
            prev->next = el;
            prev = prev->next;
        }
        return head;

    }

};

4. Deletion in a Doubly Linked List

4.1. Deletion of the head

class Solution {
public:
    ListNode* deleteHead(ListNode* head) {
        if(head == NULL) return head;
        if(head->next == NULL) return NULL;
        ListNode* temp = head;
        head = head->next;
        head->prev = NULL;
        temp->next = NULL;
        delete temp;
        return head;
    }
};

4.2. Deletion of the tail

class Solution {
public:
    ListNode* deleteTail(ListNode* head) {
        if(head == NULL || head->next == NULL) return NULL;
        ListNode* tail = head;
        while(tail->next != NULL){
            tail = tail->next;
        }
        ListNode* newTail = tail->prev;
        newTail->next = NULL;
        tail->prev = NULL;
        delete tail;
        return head;
    }
};

4.3. Deletion of the Kth element

class Solution {
public:
    ListNode* deleteKthElement(ListNode* head, int k) {
        if(head == NULL) return head;
        int count = 0;
        ListNode* kthNode = head;
        while(kthNode != NULL){
            count += 1;
            if(k == count)break;
            kthNode = kthNode->next;
        }
        ListNode* prev = kthNode->prev;
        ListNode* front = kthNode->next;

        if(prev == NULL && front == NULL){
            delete kthNode;
            return NULL;
        } 
        else if (prev == NULL){
            head = front;
            front->prev = NULL;
        }
        else if(front == NULL){
            prev->next = NULL;
        } else{
            prev->next = front;
            front->prev = prev;
        }
        delete kthNode;
        return head;
    }
};

4.4. Deletion of a given node

class Solution {
public:
    void deleteGivenNode(ListNode* node) {
        ListNode* prev = node->prev;
        ListNode* front = node->next;

        if(front == NULL){
            prev->next = NULL;
        }else{
            prev->next = front;
            front->prev = prev;
        }
    }
};

5. Insertion in a Doubly Linked List

5.1. Insertion at the head

class Solution {
public:
    ListNode* insertBeforeHead(ListNode* head, int X) {
        ListNode* temp = new ListNode(X);
        head->prev = temp;
        temp->next = head;
        return temp;
    }
};

5.2. Insertion at the tail

class Solution {
public:
    ListNode* insertBeforeTail(ListNode* head, int X) {
        if(head->next == NULL){
            ListNode* newNode = new ListNode(X, head, NULL);
            head->prev = newNode;
            return newNode;
        }

        ListNode* temp = head;
        while(temp->next != NULL){
            temp = temp->next;
        }
        ListNode* prev = temp->prev;
        ListNode* newNode = new ListNode(X, temp, prev);
        prev->next = newNode;
        temp->prev = newNode;
        return head;
    }
};

5.3. Insertion at the Kth element

class Solution {
public:
    ListNode* insertBeforeKthPosition(ListNode* head, int X, int K) {
        if(K == 1){
            ListNode* newNode = new ListNode(X, head, NULL);
            head->prev = newNode;
            return newNode;
        }
        ListNode* temp = head;
        int count = 0;
        while(temp->next != NULL){
            count += 1;
            if (count == K){
                break;
            }
            temp = temp->next;
        }
        ListNode* prev = temp->prev;
        ListNode* newNode = new ListNode(X, temp, prev);
        prev->next = newNode;
        temp->prev = newNode;

        return head;
    }
};

5.4. Insertion at a given node

class Solution {
public:
    void insertBeforeGivenNode(ListNode* node, int X) {
        ListNode* prev = node->prev;

        ListNode* newNode = new ListNode(X, node, prev);
        prev->next = newNode;
        node->prev = newNode;

    }
};
